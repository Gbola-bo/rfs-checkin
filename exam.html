// ============================================================ // RFS Assessment Portal — Google Apps Script // VERSION 3.0 // // EXAM DB SHEET STRUCTURE: // Sheet 1 "Codes": Reg Number|Last Name|First Name|Other Name|Code|Status|Used At|Type // Sheet 2 "Config": Topic|Type|Form Base URL|Entry Reg|Entry Surname|Entry FirstName|Is Open|Online Session Code // (Row 2 only — one active assessment at a time) // // CHANGES FROM V2: // - verify_exam → verify_assessment (reads Config for form URL + entry IDs) // - check_exam_open → check_assessment_open (also returns type + topic) // - set_exam_open → set_assessment_open // - Student code sheet restructured: Reg|Last|First|Other|Code|Status|UsedAt // - Config sheet: one row, drives everything (type, topic, form URL, entry IDs) // ============================================================ var EXAM_DB_ID = '1zZWIKa2imQJuRH9LnTqE1MNL_2GbYkInQhu1FTnWJAs'; var DB_SHEET_ID = '1CNr7m_8u2RQk-WoZUPR1PpHwbkKKBYn3B1r7qX6zqG8'; // new unified DB // Form URL and entry IDs are now stored in the Config sheet (row 2) // and read dynamically per assessment — no hardcoding needed. function doGet(e) { try { var action = (e.parameter.action || '').trim(); if (action === 'verify_assessment') return handleVerifyAssessment(e); if (action === 'validate_session_code') return handleValidateSessionCode(e); if (action === 'check_assessment_open') return handleCheckAssessmentOpen(e); if (action === 'set_assessment_open') return handleSetAssessmentOpen(e); return respond({ status: 'error', message: 'Unknown action.' }); } catch (err) { return respond({ status: 'error', message: 'Server error: ' + err.toString() }); } } // ── Verify Assessment ──────────────────────────────────────── // New Codes sheet cols: Reg(0)|Last(1)|First(2)|Other(3)|Code(4)|Status(5)|UsedAt(6) // New Config sheet cols: Topic(0)|Type(1)|FormBaseURL(2)|EntryReg(3)|EntrySurname(4)|EntryFirstName(5)|IsOpen(6) function handleVerifyAssessment(e) { var regNumber = (e.parameter.reg || '').trim().toUpperCase(); var examCode = (e.parameter.code || '').trim().toUpperCase(); if (!regNumber || !examCode) return respond({ status: 'error', message: 'Please enter both your registration number and access code.' }); if (!/^RFS\/\d{4}\/\d{3}$/.test(regNumber)) return respond({ status: 'error', message: 'Invalid registration number format. Use RFS/XXXX/XXX.' }); var examDB; try { examDB = SpreadsheetApp.openById(EXAM_DB_ID); } catch(err) { return respond({ status: 'error', message: 'Cannot access assessment database. Contact admin.' }); } // ── Read Config (row 2) for form URL, entry IDs, type, topic ── var configSheet = examDB.getSheetByName('Config'); if (!configSheet) return respond({ status: 'error', message: 'Config sheet not found. Contact admin.' }); var configData = configSheet.getDataRange().getValues(); if (configData.length < 2 || !configData[1][0]) return respond({ status: 'error', message: 'No assessment configured. Contact admin.' }); var cfg = configData[1]; var cfgTopic = String(cfg[0] || '').trim(); var cfgType = String(cfg[1] || 'EXAM').trim().toUpperCase(); var cfgFormBaseUrl = String(cfg[2] || '').trim(); var cfgEntryReg = String(cfg[3] || '').trim(); var cfgEntrySur = String(cfg[4] || '').trim(); var cfgEntryFirst = String(cfg[5] || '').trim(); var cfgIsOpen = String(cfg[6] || '').trim().toUpperCase() === 'TRUE'; if (!cfgIsOpen) return respond({ status: 'error', message: 'This assessment is not currently open. Please wait for admin to open it.' }); if (!cfgFormBaseUrl) return respond({ status: 'error', message: 'Assessment form not configured. Contact admin.' }); // ── Look up student code in Codes sheet ─────────────────── // Codes cols: Reg(0)|Last(1)|First(2)|Other(3)|Code(4)|Status(5)|UsedAt(6) var codesSheet = examDB.getSheetByName('Codes'); if (!codesSheet) return respond({ status: 'error', message: 'Codes sheet not found. Contact admin.' }); var codesData = codesSheet.getDataRange().getValues(); for (var i = 1; i < codesData.length; i++) { var rowReg = String(codesData[i][0]).trim().toUpperCase(); var rowCode = String(codesData[i][4]).trim().toUpperCase(); var rowStat = String(codesData[i][5]).trim().toUpperCase(); if (rowReg !== regNumber) continue; if (rowCode !== examCode) return respond({ status: 'error', message: 'Incorrect access code. Please check and try again.' }); if (rowStat === 'USED') return respond({ status: 'error', message: 'This access code has already been used. If this is an error, please contact admin.' }); // ── Lock + mark as used ─────────────────────────────────── var lock = LockService.getScriptLock(); try { lock.waitLock(10000); var freshData = codesSheet.getDataRange().getValues(); if (String(freshData[i][5]).trim().toUpperCase() === 'USED') return respond({ status: 'error', message: 'This access code has already been used. If this is an error, please contact admin.' }); var isOnlineParam = (e.parameter.isOnline || '').trim().toLowerCase() === 'true'; codesSheet.getRange(i+1, 6).setValue('USED'); // col F = Status codesSheet.getRange(i+1, 7).setValue(new Date()); // col G = UsedAt codesSheet.getRange(i+1, 8).setValue(isOnlineParam ? 'ONLINE' : 'PHYSICAL'); // col H = Type } finally { lock.releaseLock(); } // ── Name lookup from NEW DB Students tab ───────────────── var firstName = String(codesData[i][2] || '').trim(); // C=First from Codes sheet var surname = String(codesData[i][1] || '').trim(); // B=Last from Codes sheet // Try to get fresher name from Students DB try { var db = SpreadsheetApp.openById(DB_SHEET_ID); var dbSheet = db.getSheetByName('Students'); var dbData = dbSheet.getDataRange().getValues(); for (var j = 1; j < dbData.length; j++) { if (String(dbData[j][0]).trim().toUpperCase() === regNumber) { surname = String(dbData[j][1] || surname).trim(); firstName = String(dbData[j][2] || firstName).trim(); break; } } } catch(err) { /* use Codes sheet name as fallback */ } // ── Build pre-filled form URL using Config entry IDs ───── var formUrl = cfgFormBaseUrl + '?usp=pp_url' + '&' + cfgEntryReg + '=' + encodeURIComponent(regNumber) + '&' + cfgEntrySur + '=' + encodeURIComponent(surname) + '&' + cfgEntryFirst + '=' + encodeURIComponent(firstName); var encodedUrl = xorEncode(formUrl, examCode); return respond({ status : 'success', name : firstName + ' ' + surname, firstName : firstName, encodedUrl : encodedUrl, regNumber : regNumber, studentType : 'PHYSICAL', // still used for GPS check assessmentType : cfgType, topic : cfgTopic }); } return respond({ status: 'error', message: 'Registration number not found in the assessment list. Please contact admin.' }); } // ── Validate Online Session Code ───────────────────────────── // Reads Online Session Code from Config row 2, col I (index 8) // Config cols: Topic(0)|Type(1)|FormBaseURL(2)|EntryReg(3)|EntrySurname(4)| // EntryFirstName(5)|IsOpen(6)|OnlineSessionCode(7) function handleValidateSessionCode(e) { var submittedCode = (e.parameter.code || '').trim().toUpperCase(); if (!submittedCode) return respond({ status: 'error', message: 'Please enter the session code.' }); var examDB; try { examDB = SpreadsheetApp.openById(EXAM_DB_ID); } catch(err) { return respond({ status: 'error', message: 'Cannot access exam database.' }); } var configSheet = examDB.getSheetByName('Config'); if (!configSheet) return respond({ status: 'error', message: 'Config sheet not found. Contact admin.' }); var configData = configSheet.getDataRange().getValues(); // Row 2 = index 1, col I = index 7 (0-based) = Online Session Code if (configData.length < 2) return respond({ status: 'error', message: 'No assessment configured. Contact admin.' }); var sessionCode = String(configData[1][7] || '').trim().toUpperCase(); if (!sessionCode) return respond({ status: 'error', message: 'No online session code configured. Contact admin.' }); if (submittedCode !== sessionCode) return respond({ status: 'error', message: 'Incorrect session code. Please check and try again.' }); return respond({ status: 'success', message: 'Session code verified.' }); } // ── Check Assessment Open ───────────────────────────────────── // Reads Config row 2: Topic(A)|Type(B)|FormURL(C)|EntryReg(D)|EntrySur(E)|EntryFirst(F)|IsOpen(G)|SessionCode(H) function handleCheckAssessmentOpen(e) { var examDB; try { examDB = SpreadsheetApp.openById(EXAM_DB_ID); } catch(err) { return respond({ status: 'error', open: false, message: 'Cannot access exam database.' }); } var configSheet = examDB.getSheetByName('Config'); if (!configSheet) return respond({ status: 'error', open: false, message: 'Config sheet not found. Contact admin.' }); var data = configSheet.getDataRange().getValues(); if (data.length < 2 || !data[1][0]) return respond({ status: 'error', open: false, message: 'No assessment configured. Contact admin.' }); var row = data[1]; // row 2 = index 1 var topic = String(row[0] || '').trim(); // A = Topic var type = String(row[1] || 'EXAM').trim().toUpperCase(); // B = Type var isOpen = String(row[6] || '').trim().toUpperCase() === 'TRUE'; // G = Is Open return respond({ status: 'success', open: isOpen, topic: topic, assessmentType: type }); } // ── Set Assessment Open ─────────────────────────────────────── // Writes TRUE/FALSE to Config row 2 col G (Is Open = index 7, 1-based) function handleSetAssessmentOpen(e) { var value = (e.parameter.value || '').trim().toUpperCase(); if (value !== 'TRUE' && value !== 'FALSE') return respond({ status: 'error', message: 'Invalid value. Must be TRUE or FALSE.' }); var examDB; try { examDB = SpreadsheetApp.openById(EXAM_DB_ID); } catch(err) { return respond({ status: 'error', message: 'Cannot access exam database.' }); } var configSheet = examDB.getSheetByName('Config'); if (!configSheet) return respond({ status: 'error', message: 'Config sheet not found.' }); configSheet.getRange(2, 7).setValue(value); // row 2, col G = Is Open return respond({ status: 'success', open: value === 'TRUE', message: 'Assessment ' + (value === 'TRUE' ? 'opened.' : 'closed.') }); } function xorEncode(str, key) { var keyStr = key.toUpperCase(), bytes = ''; for (var i = 0; i < str.length; i++) bytes += String.fromCharCode(str.charCodeAt(i) ^ keyStr.charCodeAt(i % keyStr.length)); return Utilities.base64Encode(bytes); } function respond(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
